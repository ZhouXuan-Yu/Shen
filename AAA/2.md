# CE-CSL 手语识别训练方案（完整版）

## 一、方案概述

**数据集**: CE-CSL（连续手语数据集）  
**硬件**: RTX 5060 (8GB 显存)  
**训练方法**: 两阶段特征解耦法（ResNet特征提取 → 序列模型训练）

---

## 二、整体架构流程

```
阶段一：特征预提取
────────────────────────────────────────────────────────
原始视频 (.mp4) 
    ↓
ResNet18 (冻结权重, ImageNet预训练)
    ↓
逐帧提取特征 → (T, 512) 变长特征
    ↓
保存为 .npy 文件
    ↓
train_features/  (~2GB)
val_features/
test_features/

阶段二：序列模型训练
────────────────────────────────────────────────────────
.npy 特征 (变长序列)
    ↓
DataLoader (动态Padding + 排序)
    ↓
1D-CNN (时序卷积, 融合相邻帧, 降采样)
    ↓
BiLSTM (双向LSTM, hidden=256)
    ↓
Linear (词表大小 + 1个Blank)
    ↓
CTC Loss (对齐学习)
```

---

## 三、关键设计要点

### 3.1 变长特征处理（核心⚠️）

- **禁止固定帧数**: 不能把视频强行压缩成16帧！
- **原因**: CTC Loss 要求 `输入序列长度(T) ≥ 标签长度(L)`，强行压缩会导致模型无法收敛
- **正确做法**: 
  - 提取特征时保留原始帧数（如120帧 → (120, 512)）
  - 不同视频生成不同长度的 .npy 文件
  - 训练时通过 `collate_fn` 动态Padding对齐

### 3.2 1D-CNN + BiLSTM 架构

```
输入: (Batch, T, 512)  ← 变长特征
    ↓
permute → (Batch, 512, T)
    ↓
Conv1d(512→512, kernel=5, stride=1, padding=2)
    ↓
BatchNorm + ReLU
    ↓
MaxPool1d(kernel=2, stride=2)  ← 帧数减半
    ↓
permute → (Batch, T/2, 512)
    ↓
BiLSTM(input=512, hidden=256, num_layers=2, bidirectional=True)
    ↓
Dropout(0.5)
    ↓
Linear(256*2 → num_classes+1)  ← +1是Blank标签
    ↓
log_softmax
    ↓
permute → (T/2, Batch, num_classes+1)
    ↓
CTC Loss
```

### 3.3 pack_padded_sequence 优化

- 作用: 忽略Padding的零，提升计算效率
- 要求: 输入必须按长度从大到小排序

```python
# ============== ResNet特征提取器 ==============
class ResNetFeatureExtractor:
    def __init__(self, model_name="resnet18"):
        # 加载预训练模型
        if model_name == "resnet18":
            resnet = models.resnet18(pretrained=True)
        elif model_name == "resnet50":
            resnet = models.resnet50(pretrained=True)
        else:
            raise ValueError(f"不支持的模型: {model_name}")
        
        # 移除最后全连接层，保留特征提取部分
        self.backbone = nn.Sequential(*list(resnet.children())[:-1])
        self.backbone.eval()  # 评估模式
        
        # 冻结所有参数
        for param in self.backbone.parameters():
            param.requires_grad = False
        
        self.backbone.to(cfg.DEVICE)
        
    def extract(self, frames):
        """
        提取视频帧特征
        Args:
            frames: list of numpy arrays, shape (H, W, C), BGR format
        Returns:
            features: numpy array, shape (T, 512)
        """
        # 预处理
        processed = []
        for frame in frames:
            # BGR → RGB
            frame = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
            # 缩放 + 中心裁剪
            frame = cv2.resize(frame, (cfg.IMAGE_SIZE + 32, cfg.IMAGE_SIZE + 32))
            h, w = frame.shape[:2]
            start_h = (h - cfg.IMAGE_SIZE) // 2
            start_w = (w - cfg.IMAGE_SIZE) // 2
            frame = frame[start_h:start_h+cfg.IMAGE_SIZE, start_w:start_w+cfg.IMAGE_SIZE]
            # 归一化 (ImageNet统计量)
            frame = frame.astype(np.float32) / 255.0
            frame = (frame - np.array([0.485, 0.456, 0.406])) / np.array([0.229, 0.224, 0.225])
            # CHW
            frame = frame.transpose(2, 0, 1)
            processed.append(frame)
        
        # 转为Tensor
        batch = torch.FloatTensor(np.stack(processed))
        
        # 提取特征
        with torch.no_grad():
            features = self.backbone(batch.to(cfg.DEVICE))
            features = features.squeeze(-1).squeeze(-1)  # (T, 512)
        
        return features.cpu().numpy()

# ============== 视频加载 ==============
def load_video(video_path, max_frames=None):
    """加载视频并返回帧列表"""
    cap = cv2.VideoCapture(video_path)
    frames = []
    
    frame_idx = 0
    while True:
        ret, frame = cap.read()
        if not ret:
            break
        if frame_idx % cfg.FRAME_SAMPLE_RATE == 0:
            frames.append(frame)
        frame_idx += 1
        
        if max_frames and len(frames) >= max_frames:
            break
    
    cap.release()
    return frames

def get_video_path(video_dir, split, translator_id, video_num):
    """构建视频文件路径"""
    # 格式: video_dir/split/translator_id/video_num.mp4
    # 例如: video/train/A/0001.mp4
    video_path = os.path.join(video_dir, split, translator_id, f"{video_num:04d}.mp4")
    return video_path

# ============== 主提取流程 ==============
def extract_split_features(extractor, split="train"):
    """提取单个数据集分割的特征"""
    
    # 读取标签文件
    label_file = os.path.join(cfg.LABEL_DIR, f"{split}.csv")
    df = pd.read_csv(label_file)
    
    # 创建输出目录
    if split == "train":

    elif split == "val":
        output_dir = cfg.VAL_FEATURES
    elif split == "test":
        output_dir = cfg.TEST_FEATURES
    else:
        raise ValueError(f"未知的split: {split}")
    
    os.makedirs(output_dir, exist_ok=True)
    
    print(f"\n开始提取 {split} 集特征，共 {len(df)} 个视频...")
    
    for idx, row in tqdm(df.iterrows(), total=len(df)):
        video_num = row["video_number"]
        translator_id = row["translator_id"]  # A-L
        
        # 构建视频路径
        video_path = get_video_path(cfg.VIDEO_DIR, split, translator_id, video_num)
        
        # 输出特征路径
        feature_path = os.path.join(output_dir, f"{video_num:04d}_{translator_id}.npy")
        
        # 如果已存在则跳过
        if os.path.exists(feature_path):
            continue
        
        # 加载视频
        if not os.path.exists(video_path):
            print(f"警告: 视频不存在 {video_path}")
            continue
            
        frames = load_video(video_path)
        
        if len(frames) == 0:
            print(f"警告: 视频为空 {video_path}")
            continue
        
        # 提取特征
        features = extractor.extract(frames)
        
        # 保存
        np.save(feature_path, features)
    
    print(f"{split} 集特征提取完成！")

# ============== 入口 ==============
if __name__ == "__main__":
```

```python
"""

import numpy as np

from torch.utils.data import Dataset, DataLoader

class CSLDataset(Dataset):
    """CE-CSL手语数据集"""
    
    def __init__(self, features_dir, label_csv, vocab, split="train"):
        """
        Args:
            features_dir: 特征文件目录 (train_features/)
            label_csv: 标签CSV文件路径
            vocab: Vocabulary对象
            split: 数据集分割
        """
        self.features_dir = features_dir
        self.vocab = vocab
        self.split = split
        
        # 读取标签
        df = pd.read_csv(label_csv)
        
        # 构建样本列表
        self.samples = []
        for idx, row in df.iterrows():
            video_num = row["video_number"]
            translator_id = row["translator_id"]
            gloss_text = row["gloss"]  # 手语词序列
            
            # 特征文件路径
            feature_file = f"{video_num:04d}_{translator_id}.npy"
            feature_path = os.path.join(features_dir, feature_file)
            
            if not os.path.exists(feature_path):
                continue
            
            # 解析gloss序列
            gloss_ids = self.vocab.encode(gloss_text)
            
            self.samples.append({
                "feature_path": feature_path,
                "gloss_ids": gloss_ids,
                "chinese": row.get("chinese", ""),
                "video_num": video_num,
                "translator_id": translator_id
            })
    
    def __len__(self):
        return len(self.samples)
    
    def __getitem__(self, idx):
        sample = self.samples[idx]
        
        # 加载特征 (T, 512)
        features = np.load(sample["feature_path"])
        features = torch.FloatTensor(features)
        
        # 标签
        gloss_ids = torch.LongTensor(sample["gloss_ids"])
        
        return features, gloss_ids


def collate_fn(batch):
    """
    批处理整理函数（核心⚠️）
    - 按特征长度从大到小排序（pack_padded_sequence要求）
    - 动态Padding对齐
    - 拼接标签
    """
    # 排序（从大到小）
    batch.sort(key=lambda x: x[0].shape[0], reverse=True)
    
    features, labels = zip(*batch)
    
    # 1. 特征长度序列
    feature_lengths = torch.LongTensor([f.shape[0] for f in features])
    
    # 2. 标签长度序列
    label_lengths = torch.LongTensor([len(l) for l in labels])
    
    # 3. Padding对齐特征
    padded_features = torch.nn.utils.rnn.pad_sequence(features, batch_first=True)
    # 形状: (Batch, Max_T, 512)
    
    # 4. 拼接所有标签（CTC要求1D连续张量）
    labels_concat = torch.cat(labels)
    
    return {
        "features": padded_features,
        "labels": labels_concat,
        "feature_lengths": feature_lengths,
        "label_lengths": label_lengths
    }


class Vocabulary:
    """词表管理"""
    
    def __init__(self):
        self.token2id = {"<blank>": 0, "<unk>": 1}
        self.id2token = {0: "<blank>", 1: "<unk>"}
        self.n_tokens = 2
    
    def build_vocab(self, gloss_sequences):
        """从gloss序列构建词表"""
        for seq in gloss_sequences:
            for token in seq:
                if token not in self.token2id:
                    self.token2id[token] = self.n_tokens
                    self.id2token[self.n_tokens] = token
                    self.n_tokens += 1
    
    def encode(self, text):
        """文本转ID"""
        return [self.token2id.get(t, self.token2id["<unk>"]) for t in text]
    
    def decode(self, ids):
        """ID转文本"""
        return [self.id2token.get(i, "<unk>") for i in ids]
```

### 4.3 1D-CNN + BiLSTM 模型 (model.py)

```python
"""
阶段二：时序分类模型
架构：1D-CNN + BiLSTM + CTC
"""
import torch
import torch.nn as nn
import torch.nn.functional as F

class TemporalConvBiLSTM(nn.Module):
    """1D-CNN + BiLSTM 时序分类模型"""
    
    def __init__(self, input_size=512, hidden_size=256, num_layers=2, 
                 num_classes=1000, dropout=0.5):
        super(TemporalConvBiLSTM, self).__init__()
        
        self.input_size = input_size
        self.hidden_size = hidden_size
        self.num_classes = num_classes  # 词表大小（不含blank）
        
        # ========== 1D-CNN 时序卷积层 ==========
        # 作用：融合相邻帧特征，降采样时间维度
        self.temporal_conv = nn.Sequential(
            # Conv1d: 输入( Batch, Channels, T)
            nn.Conv1d(in_channels=input_size, 
                      out_channels=input_size, 
                      kernel_size=5, 
                      stride=1, 
                      padding=2),  # padding=2 保持T不变
            nn.BatchNorm1d(input_size),
            nn.ReLU(inplace=True),
            nn.Dropout(dropout),
            # MaxPool1d: 将T减半
            nn.MaxPool1d(kernel_size=2, stride=2)
        )
        
        # ========== BiLSTM ==========
        self.lstm = nn.LSTM(
            input_size=input_size,
            hidden_size=hidden_size,
            num_layers=num_layers,
            batch_first=True,
            bidirectional=True,
            dropout=dropout if num_layers > 1 else 0
        )
        
        # ========== 分类层 ==========
        # 输出: 词表大小 + 1个Blank标签
        self.fc = nn.Linear(hidden_size * 2, num_classes + 1)
        
        # 日志softmax
        self.log_softmax = nn.LogSoftmax(dim=2)
    
    def forward(self, x, feature_lengths):
        """
        Args:
            x: (Batch, T, 512) 变长特征
            feature_lengths: (Batch,) 原始特征长度
        Returns:
            outputs: (T', Batch, num_classes+1) CTC输入
            output_lengths: (Batch,) 降采样后的长度
        """
        # Conv1d 需要 (Batch, Channels, T)
        x = x.permute(0, 2, 1)  # (B, 512, T)
        
        # 1D卷积
        x = self.temporal_conv(x)  # (B, 512, T/2)
        
        # 恢复 (Batch, T', 512)
        x = x.permute(0, 2, 1)
        
        # 计算卷积后的实际长度（MaxPool1d: T' = T//2）
        output_lengths = feature_lengths // 2
        
        # Pack Padded Sequence（忽略Padding的零）
        packed_x = nn.utils.rnn.pack_padded_sequence(
            x, output_lengths.cpu(), batch_first=True, enforce_sorted=True
        )
        
        # LSTM
        packed_out, _ = self.lstm(packed_x)
        
        # 解压
        lstm_out, _ = nn.utils.rnn.pad_packed_sequence(packed_out, batch_first=True)
        # lstm_out: (Batch, T', hidden*2)
        
        # 分类
        out = self.fc(lstm_out)  # (Batch, T', num_classes+1)
        
        # CTC要求: (T', Batch, num_classes+1)
        out = out.permute(1, 0, 2)
        
        # LogSoftmax
        out = self.log_softmax(out)
        
        return out, output_lengths

            features = batch["features"].to(self.device)
            labels = batch["labels"].to(self.device)
            feature_lengths = batch["feature_lengths"].to(self.device)
            label_lengths = batch["label_lengths"]
            
            # 前向
            outputs, output_lengths = self.model(features, feature_lengths)
            
            # CTC Loss
            loss = self.criterion(outputs, labels, output_lengths, label_lengths)

            self.optimizer.zero_grad()

    @torch.no_grad()
    def greedy_decode(self, outputs):
        """
        贪婪解码
        Args:
            outputs: (T, B, C) log_probs
        Returns:
            predictions: list of id sequences
        """
        # 获取最大概率的类别
        predictions = outputs.argmax(dim=2)  # (T, B)
        predictions = predictions.permute(1, 0)  # (B, T)
        
        decoded = []
        for pred in predictions:
            # 移除连续重复和空白标签
            decoded_seq = []
            prev = -1
            for idx in pred.tolist():
                if idx != 0 and idx != prev:  # 0是blank
                    decoded_seq.append(idx)
                prev = idx
            decoded.append(decoded_seq)
        
        return decoded
    
    @torch.no_grad()

            label_lengths = batch["label_lengths"]
```

```python
        batch_size=cfg.BATCH_SIZE, 
        shuffle=False, 
        collate_fn=collate_fn,
        num_workers=4
    )
    
    print(f"训练集: {len(train_dataset)} 样本")
    print(f"验证集: {len(val_dataset)} 样本")
    
    # 4. 创建模型
    model = TemporalConvBiLSTM(
        input_size=cfg.INPUT_SIZE,
        hidden_size=cfg.HIDDEN_SIZE,
        num_layers=cfg.NUM_LAYERS,
        num_classes=vocab.n_tokens,
        dropout=cfg.DROPOUT
    )
    
    # 5. 训练
    trainer = CTCTrainer(model, vocab, cfg.DEVICE, lr=cfg.LR)
    
    best_loss = float("inf")
    
    for epoch in range(cfg.EPOCHS):
        # 训练
        train_loss = trainer.train_epoch(train_loader)
        
        # 验证
        val_loss, _ = trainer.evaluate(val_loader)
        
        print(f"Epoch {epoch+1}/{cfg.EPOCHS} | "
              f"Train Loss: {train_loss:.4f} | "
              f"Val Loss: {val_loss:.4f}")
        
        # 保存最佳模型
        if val_loss < best_loss:
            best_loss = val_loss
            save_path = os.path.join(cfg.OUTPUT_DIR, "best_model.pt")
            torch.save({
                "epoch": epoch,
                "model_state_dict": model.state_dict(),
                "vocab": vocab,
                "config": {
                    "input_size": cfg.INPUT_SIZE,
                    "hidden_size": cfg.HIDDEN_SIZE,
                    "num_layers": cfg.NUM_LAYERS,
                    "num_classes": vocab.n_tokens
                }
            }, save_path)
            print(f"保存最佳模型: {save_path}")
    
    print("训练完成!")

if __name__ == "__main__":
    main()
```
